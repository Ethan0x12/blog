## 技术栈概览

- **Next.js 16.0.0**：React框架，支持SSR/SSG，提供优秀的开发体验和性能优化
- **React 19.2.0**：JavaScript库，用于构建用户界面
- **TailwindCSS 4.0.0**：实用优先的CSS框架，用于快速构建现代UI
- **Ethers 6.15.0**：与以太坊区块链交互的JavaScript库
- **Wagmi 2.18.1**：React Hooks库，简化以太坊交互
- **RainbowKit 2.2.9**：钱包连接UI库，提供优雅的钱包选择体验

## 开发环境搭建

在开始项目之前，我们需要搭建基础的开发环境：

```bash

npx create-next-app@latest web3-starter-project --typescript --tailwind --eslint

# 进入项目目录
cd web3-starter-project

# 安装Web3相关依赖
npm install ethers@6.15.0 wagmi@2.18.1 @rainbow-me/rainbowkit@2.2.9 viem
```

## 项目一：钱包连接演示

### 功能概述

实现基本的钱包连接功能，显示连接的钱包地址、链ID、账户余额等信息。

### 配置Wagmi和RainbowKit

创建`src/app/providers.tsx`文件：

```tsx
'use client';

import { WagmiProvider } from 'wagmi';
import { RainbowKitProvider } from '@rainbow-me/rainbowkit';
import { sepolia, mainnet } from 'wagmi/chains';
import { createConfig, http } from 'wagmi';
import { ConnectButton } from '@rainbow-me/rainbowkit';
import '@rainbow-me/rainbowkit/styles.css';
import { QueryClient, QueryClientProvider } from '@tanstack/react-query';

// 创建QueryClient实例
const queryClient = new QueryClient();

// 配置Wagmi客户端
const config = createConfig({
  chains: [mainnet, sepolia],
  transports: {
    [mainnet.id]: http(`https://mainnet.infura.io/v3/${process.env.NEXT_PUBLIC_INFURA_API_KEY}`),
    [sepolia.id]: http(`https://sepolia.infura.io/v3/${process.env.NEXT_PUBLIC_INFURA_API_KEY}`),
  },
});

export function Web3Providers({ children }: { children: React.ReactNode }) {
  return (
    <WagmiProvider config={config}>
      <QueryClientProvider client={queryClient}>
        <RainbowKitProvider initialChainId={sepolia.id}>
          {children}
        </RainbowKitProvider>
      </QueryClientProvider>
    </WagmiProvider>
  );
}
```

## 项目二：代币转账DApp

### 功能概述

创建一个简单的代币转账界面，支持ETH转账和ERC-20代币转账。

### 创建代币转账组件

创建`src/components/TokenTransfer.tsx`文件：

```tsx
'use client';

import { useState } from 'react';
import { useAccount, useSendTransaction, useWriteContract } from 'wagmi';
import { parseEther } from 'ethers';
import { erc20Abi } from 'viem';

export function TokenTransfer() {
  const { address, isConnected } = useAccount();
  
  // 表单状态
  const [recipient, setRecipient] = useState('');
  const [amount, setAmount] = useState('');
  const [tokenAddress, setTokenAddress] = useState('');
  const [isTokenTransfer, setIsTokenTransfer] = useState(false);
  const [isSubmitting, setIsSubmitting] = useState(false);
  
  // ETH转账
  const { sendTransaction } = useSendTransaction();
  
  // ERC-20代币转账
  const { writeContract } = useWriteContract();

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    
    if (!isConnected || !address) {
      alert('请先连接钱包');
      return;
    }
    
    if (!recipient || !amount) {
      alert('请填写收款地址和转账金额');
      return;
    }
    
    if (isTokenTransfer && !tokenAddress) {
      alert('请填写代币合约地址');
      return;
    }
    
    setIsSubmitting(true);
    
    try {
      if (isTokenTransfer) {
        // ERC-20代币转账
        writeContract({
          address: tokenAddress as `0x${string}`,
          abi: erc20Abi,
          functionName: 'transfer',
          args: [recipient as `0x${string}`, parseEther(amount)],
        });
      } else {
        // ETH转账
        sendTransaction({
          to: recipient as `0x${string}`,
          value: parseEther(amount),
        });
      }
    } catch (error) {
      console.error('转账失败:', error);
      alert('转账失败，请检查输入并确认交易');
    } finally {
      setIsSubmitting(false);
    }
  };

  if (!isConnected) {
    return (
      <div className="max-w-md mx-auto p-6 bg-gray-50 dark:bg-gray-900 rounded-lg shadow-lg text-center py-12">
        <p className="text-gray-500 dark:text-gray-400">请连接钱包以使用转账功能</p>
      </div>
    );
  }

  return (
    <div className="max-w-md mx-auto p-6 bg-gray-50 dark:bg-gray-900 rounded-lg shadow-lg">
      <h2 className="text-2xl font-bold mb-6 text-center text-gray-900 dark:text-white">
        代币转账
      </h2>
      
      <div className="mb-4 flex gap-2">
        <label className="flex items-center gap-2 cursor-pointer">
          <input 
            type="radio" 
            checked={!isTokenTransfer} 
            onChange={() => setIsTokenTransfer(false)}
            className="text-blue-500 focus:ring-blue-500 border-gray-300 dark:border-gray-600"
          />
          <span className="text-gray-700 dark:text-gray-300">ETH转账</span>
        </label>
        
        <label className="flex items-center gap-2 cursor-pointer">
          <input 
            type="radio" 
            checked={isTokenTransfer} 
            onChange={() => setIsTokenTransfer(true)}
            className="text-blue-500 focus:ring-blue-500 border-gray-300 dark:border-gray-600"
          />
          <span className="text-gray-700 dark:text-gray-300">ERC-20转账</span>
        </label>
      </div>
      
      <form onSubmit={handleSubmit} className="space-y-4">
        {isTokenTransfer && (
          <div>
            <label htmlFor="tokenAddress" className="block text-sm font-medium text-gray-700 dark:text-gray-300 mb-1">
              代币合约地址
            </label>
            <input
              type="text"
              id="tokenAddress"
              value={tokenAddress}
              onChange={(e) => setTokenAddress(e.target.value)}
              placeholder="0x..."
              className="w-full px-3 py-2 border border-gray-300 dark:border-gray-600 rounded-md shadow-sm focus:outline-none focus:ring-blue-500 dark:bg-gray-700 dark:text-white"
            />
          </div>
        )}
        
        <div>
          <label htmlFor="recipient" className="block text-sm font-medium text-gray-700 dark:text-gray-300 mb-1">
            收款地址
          </label>
          <input
            type="text"
            id="recipient"
            value={recipient}
            onChange={(e) => setRecipient(e.target.value)}
            placeholder="0x..."
            className="w-full px-3 py-2 border border-gray-300 dark:border-gray-600 rounded-md shadow-sm focus:outline-none focus:ring-blue-500 dark:bg-gray-700 dark:text-white"
          />
        </div>
        
        <div>
          <label htmlFor="amount" className="block text-sm font-medium text-gray-700 dark:text-gray-300 mb-1">
            转账金额
          </label>
          <input
            type="text"
            id="amount"
            value={amount}
            onChange={(e) => setAmount(e.target.value)}
            placeholder={isTokenTransfer ? "代币数量" : "ETH数量"}
            className="w-full px-3 py-2 border border-gray-300 dark:border-gray-600 rounded-md shadow-sm focus:outline-none focus:ring-blue-500 dark:bg-gray-700 dark:text-white"
          />
        </div>
        
        <button
          type="submit"
          disabled={isSubmitting}
          className="w-full bg-blue-500 hover:bg-blue-600 text-white font-medium py-2 px-4 rounded-md transition-colors duration-200 focus:outline-none focus:ring-2 focus:ring-offset-2 focus:ring-blue-500 disabled:opacity-50 disabled:cursor-not-allowed"
        >
          {isSubmitting ? '处理中...' : '确认转账'}
        </button>
      </form>
    </div>
  );
}
```

### 创建NFT数据处理工具

创建`src/utils/nftUtils.ts`文件：

```typescript
import axios from 'axios';

export interface NFT {
  tokenId: string;
  tokenURI: string;
  contractAddress: string;
  metadata?: NFTMetadata;
  imageUrl?: string;
  name?: string;
  description?: string;
}

export interface NFTMetadata {
  name: string;
  description: string;
  image: string;
  attributes?: Array<{
    trait_type: string;
    value: string | number;
  }>;
}

// 将IPFS URI转换为HTTP URL
export const ipfsToHttp = (ipfsUri: string): string => {
  if (ipfsUri.startsWith('ipfs://')) {
    return `https://ipfs.io/ipfs/${ipfsUri.replace('ipfs://', '')}`;
  }
  return ipfsUri;
};

// 获取NFT元数据
export const fetchNFTMetadata = async (tokenURI: string): Promise<NFTMetadata | null> => {
  try {
    const httpUrl = ipfsToHttp(tokenURI);
    const response = await axios.get(httpUrl);
    return response.data;
  } catch (error) {
    console.error('获取NFT元数据失败:', error);
    return null;
  }
};

// 格式化NFT数据
export const formatNFTs = async (nfts: NFT[]): Promise<NFT[]> => {
  const formattedNFTs = await Promise.all(
    nfts.map(async (nft) => {
      if (!nft.tokenURI) return nft;
      
      try {
        const metadata = await fetchNFTMetadata(nft.tokenURI);
        if (metadata) {
          return {
            ...nft,
            metadata,
            name: metadata.name,
            description: metadata.description,
            imageUrl: ipfsToHttp(metadata.image)
          };
        }
      } catch (error) {
        console.error(`处理NFT ${nft.tokenId} 的元数据时出错:`, error);
      }
      
      return nft;
    })
  );
  
  return formattedNFTs.filter(nft => nft.imageUrl);
};
```

```jsx
// NativeWalletConnect.jsx
import React, { useState, useEffect } from 'react';
import { ethers } from 'ethers';

const NativeWalletConnect = () => {
  const [connected, setConnected] = useState(false);
  const [account, setAccount] = useState('');
  const [balance, setBalance] = useState('');
  const [error, setError] = useState('');

  // 检查MetaMask是否安装
  const checkMetaMask = () => {
    if (typeof window.ethereum === 'undefined') {
      setError('请安装MetaMask钱包');
      return false;
    }
    return true;
  };

  // 连接钱包
  const connectWallet = async () => {
    if (!checkMetaMask()) return;

    try {
      setError('');
      const accounts = await window.ethereum.request({
        method: 'eth_requestAccounts'
      });
      setAccount(accounts[0]);
      setConnected(true);
      
      // 获取余额
      const provider = new ethers.providers.Web3Provider(window.ethereum);
      const bal = await provider.getBalance(accounts[0]);
      setBalance(ethers.utils.formatEther(bal));
    } catch (err) {
      setError('连接失败: ' + err.message);
    }
  };

  // 断开连接
  const disconnectWallet = () => {
    setConnected(false);
    setAccount('');
    setBalance('');
  };

  // 监听账户变化
  useEffect(() => {
    if (typeof window.ethereum !== 'undefined') {
      window.ethereum.on('accountsChanged', (accounts) => {
        if (accounts.length > 0) {
          setAccount(accounts[0]);

          provider.getBalance(accounts[0]).then(bal => {
            setBalance(ethers.utils.formatEther(bal));
          });
        } else {
          disconnectWallet();
        }
      });

      return () => {
        window.ethereum.removeListener('accountsChanged', () => {});
      };
    }
  }, []);

  return (
    <div className="native-wallet-connect">
      {error && <div className="error-message">{error}</div>}
      
      {!connected ? (
        <button onClick={connectWallet} className="connect-button">
```

## 代币转账DApp

代币转账是Web3应用中最常见的功能之一，本部分将实现一个简单的ERC-20代币转账应用。

### 功能需求

- 显示用户的代币余额
- 输入目标地址和转账金额
- 执行转账操作
- 显示交易状态和结果

### 完整实现

```jsx
// TokenTransfer.jsx
import React, { useState, useEffect } from 'react';
import { ethers } from 'ethers';

// ERC-20代币ABI（简化版）
const ERC20_ABI = [
  'function balanceOf(address owner) view returns (uint256)',
  'function transfer(address to, uint256 amount) returns (bool)',
  'function decimals() view returns (uint8)'
];

// 代币地址示例（USDC在Goerli测试网）
const TOKEN_ADDRESS = '0x07865c6e87b9f70255377e024ace6630c1eaa37f';

const TokenTransfer = () => {
  const [provider, setProvider] = useState(null);
  const [signer, setSigner] = useState(null);
  const [tokenContract, setTokenContract] = useState(null);
  const [userAddress, setUserAddress] = useState('');
  const [balance, setBalance] = useState('0');
  const [decimals, setDecimals] = useState(18);
  const [recipient, setRecipient] = useState('');
  const [amount, setAmount] = useState('');
  const [isLoading, setIsLoading] = useState(false);
  const [transactionHash, setTransactionHash] = useState('');
  const [error, setError] = useState('');
  const [success, setSuccess] = useState(false);

  // 初始化连接
  useEffect(() => {
    const init = async () => {
      if (typeof window.ethereum !== 'undefined') {
        try {
          // 创建Provider和Signer
          const web3Provider = new ethers.providers.Web3Provider(window.ethereum);
          setProvider(web3Provider);
          
          const accounts = await web3Provider.listAccounts();
          if (accounts.length > 0) {
            setUserAddress(accounts[0]);
            const web3Signer = web3Provider.getSigner();
            setSigner(web3Signer);
            
            // 创建代币合约实例
            const contract = new ethers.Contract(TOKEN_ADDRESS, ERC20_ABI, web3Signer);
            setTokenContract(contract);
            
            // 获取代币精度
            const tokenDecimals = await contract.decimals();
            setDecimals(tokenDecimals);
            
            // 获取余额
            await fetchBalance(accounts[0], contract, tokenDecimals);
          }
        } catch (err) {
          console.error('初始化错误:', err);
          setError('连接钱包失败');
        }
      }
    };
    
    init();
    
    // 监听账户变化
    if (typeof window.ethereum !== 'undefined') {
      window.ethereum.on('accountsChanged', handleAccountsChanged);
      return () => {
        window.ethereum.removeListener('accountsChanged', handleAccountsChanged);
      };
    }
  }, []);

  // 处理账户变化
  const handleAccountsChanged = async (accounts) => {
    if (accounts.length > 0) {
      setUserAddress(accounts[0]);
      setError('');
      
      if (provider) {
        const web3Signer = provider.getSigner();
        setSigner(web3Signer);
        
        const contract = new ethers.Contract(TOKEN_ADDRESS, ERC20_ABI, web3Signer);
        setTokenContract(contract);
        
        await fetchBalance(accounts[0], contract, decimals);
      }
    } else {
      setUserAddress('');
      setBalance('0');
    }
  };

  // 获取代币余额
  const fetchBalance = async (address, contract, tokenDecimals) => {
    if (!contract || !address) return;
    
    try {
      const bal = await contract.balanceOf(address);
      setBalance(ethers.utils.formatUnits(bal, tokenDecimals));
    } catch (err) {
      console.error('获取余额失败:', err);
      setError('获取余额失败');
    }
  };

  // 连接钱包
  const connectWallet = async () => {
    if (typeof window.ethereum === 'undefined') {
      setError('请安装MetaMask钱包');
      return;
    }
    
    try {
      setError('');
      const accounts = await window.ethereum.request({
        method: 'eth_requestAccounts'
      });
      handleAccountsChanged(accounts);
    } catch (err) {
      console.error('连接失败:', err);
      setError('连接钱包失败');
    }
  };

  // 处理转账
  const handleTransfer = async (e) => {
    e.preventDefault();
    
    if (!tokenContract || !recipient || !amount) {
      setError('请填写完整信息');
      return;
    }
    
    // 验证地址格式
    if (!ethers.utils.isAddress(recipient)) {
      setError('请输入有效的以太坊地址');
      return;
    }
    
    setIsLoading(true);
    setError('');
    setSuccess(false);
    setTransactionHash('');
    
    try {
      // 转换金额为合约可接受的格式（乘以精度）
      const amountInWei = ethers.utils.parseUnits(amount, decimals);
      
      // 执行转账
      const tx = await tokenContract.transfer(recipient, amountInWei);
      setTransactionHash(tx.hash);
      
      // 等待交易确认
      await tx.wait();
      
      setSuccess(true);
      setAmount('');
      
      // 更新余额
      await fetchBalance(userAddress, tokenContract, decimals);
    } catch (err) {
      console.error('转账失败:', err);
      setError('转账失败: ' + (err.reason || err.message));
    } finally {
      setIsLoading(false);
    }
  };

  return (
    <div className="token-transfer">
      <h2>代币转账DApp</h2>
      
      {!userAddress ? (
        <button onClick={connectWallet} className="connect-button">
          连接钱包
        </button>
      ) : (
        <div className="transfer-container">
          <div className="wallet-info">
            <p>当前地址: {userAddress.substring(0, 6)}...{userAddress.substring(userAddress.length - 4)}</p>
            <p>代币余额: {balance}</p>
          </div>
          
          <form onSubmit={handleTransfer} className="transfer-form">
            <div className="form-group">
              <label htmlFor="recipient">接收地址</label>
              <input
                type="text"
                id="recipient"
                value={recipient}
                onChange={(e) => setRecipient(e.target.value)}
                placeholder="0x..."
                required
              />
            </div>
            
            <div className="form-group">
              <label htmlFor="amount">转账金额</label>
              <input
                type="number"
                id="amount"
                value={amount}
                onChange={(e) => setAmount(e.target.value)}
                placeholder="0.0"
                step="0.000001"
                min="0"
                required
              />
            </div>
            
            <button 
              type="submit" 
              className="transfer-button"
              disabled={isLoading}
            >
              {isLoading ? '处理中...' : '转账'}
            </button>
          </form>
          
          {error && <div className="error-message">{error}</div>}
          
          {success && (
            <div className="success-message">
              <p>转账成功！</p>
              <p>
                交易哈希: 
                <a 
                  href={`https://goerli.etherscan.io/tx/${transactionHash}`}
                  target="_blank"
                  rel="noopener noreferrer"
                >
                  {transactionHash.substring(0, 10)}...
                </a>
              </p>
            </div>
          )}
        </div>
      )}
    </div>
  );
};

export default TokenTransfer;
```

```jsx
                    <p>ID: {nft.tokenId}</p>
                  </div>
                </div>
              ))}
            </div>
          )}
        </div>
      )}
      
      {/* NFT详情模态框 */}
      {selectedNft && (
        <div className="modal-overlay" onClick={closeNftDetails}>
          <div className="modal-content" onClick={(e) => e.stopPropagation()}>
            <button className="close-button" onClick={closeNftDetails}>&times;</button>
            <div className="nft-detail-image">
              <img src={selectedNft.imageUrl} alt={selectedNft.name} />
            </div>
            <div className="nft-detail-info">
              <h2>{selectedNft.name}</h2>
              <p>{selectedNft.description}</p>
              <div className="detail-meta">
                <p><strong>合约地址:</strong> {selectedNft.contractAddress}</p>
                <p><strong>代币ID:</strong> {selectedNft.tokenId}</p>
                <a 
                  href={`https://opensea.io/assets/${selectedNft.contractAddress}/${selectedNft.tokenId}`}
                  target="_blank"
                  rel="noopener noreferrer"
                  className="opensea-link"
                >
                  在OpenSea上查看 <FaExternalLinkAlt className="icon" />
                </a>
              </div>
            </div>
          </div>
        </div>
      )}
    </div>
  );
};

export default NFTViewer;
```

### 使用Alchemy NFT API获取真实NFT数据

```jsx
// 替换fetchUserNFTs函数
const fetchUserNFTs = async (address, network) => {
  setIsLoading(true);
  setError('');
  
  // Alchemy API密钥（实际项目中应该从环境变量获取）
  const ALCHEMY_API_KEY = 'your_alchemy_api_key';
  
  // 根据网络选择不同的Alchemy端点
  const baseUrl = network === 'polygon' 
    ? `https://polygon-mainnet.g.alchemy.com/nft/v2/${ALCHEMY_API_KEY}`
    : network === 'goerli'
      ? `https://eth-goerli.g.alchemy.com/nft/v2/${ALCHEMY_API_KEY}`
      : `https://eth-mainnet.g.alchemy.com/nft/v2/${ALCHEMY_API_KEY}`;
  
  try {
    // 使用Alchemy API获取NFTs
    const response = await fetch(`${baseUrl}/getNFTs?owner=${address}&withMetadata=true`);
    const data = await response.json();
    
    // 处理NFT数据
    const formattedNfts = data.ownedNfts.map((nft, index) => {
      const imageUrl = nft.media && nft.media.length > 0 
        ? nft.media[0].gateway 
        : 'https://via.placeholder.com/300x300?text=No+Image';
      
      return {
        id: index.toString(),
        name: nft.title || 'Unknown NFT',
        imageUrl: imageUrl,
        contractAddress: nft.contract.address,
        tokenId: nft.id.tokenId,
        description: nft.description || 'No description available'
      };
    });
    
    setNfts(formattedNfts);
  } catch (err) {
    console.error('获取NFT失败:', err);
    setError('获取NFT失败，请稍后重试');
  } finally {
    setIsLoading(false);
  }
};
```
